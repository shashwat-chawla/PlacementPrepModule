class Solution {
public:
    int removeDuplicates(vector<int>& nums) {
        int n=nums.size();
       int i=0,j=1;
        for( j=1;j<n;j++)
        {
            if(nums[i]!=nums[j])
            {
                nums[++i]=nums[j];
            }
        }
       // nums[i]=nums[j-1];
        return i+1;
        
    }
};
